44constexpr char explicitPopcount(uint32_t i)
45{
46#warning NOT USING BUILTIN POPCOUNT
47 //Magic code that compiles to popcnt https://stackoverflow.com/questions/109023/count-the-number-of-set-bits-in-a-32-bit-integer
48 i = i - ((i >> 1) & 0x55555555); // add pairs of bits
49 i = (i & 0x33333333) + ((i >> 2) & 0x33333333); // quads
50 i = (i + (i >> 4)) & 0x0F0F0F0F; // groups of 8
51 i *= 0x01010101; // horizontal sum of bytes
52 return i >> 24; // return just that top byte (after truncating to 32-bit even when int is wider than uint32_t)
53}
