83 template<typename VectorType>
84 static void compressVector(const vectorView<const Matrix<VectorType>,Eigen::RowMajor>& src, vectorView<Matrix<VectorType>,Eigen::RowMajor> dst,
85 std::shared_ptr<compressor> thisPtr)
86 {
87 std::shared_ptr<compressor> srcCompressor;
88 if (src.getIsCompressed(srcCompressor)) // this comes up often enough that its better to special case it than throw an error
89 {
90 if (srcCompressor.get() == thisPtr.get())
91 {
92 dst.copy(src);
93 return;
94 }
95 __builtin_trap();
96 }
97
98 dst.resize(thisPtr->decompressPerm.size(),true,thisPtr,false);
99 for (size_t i = 0; i < thisPtr->compressPerm.size(); i++)
100 {
101 int64_t newIdx = thisPtr->compressPerm[i];
102 if (newIdx >= 0)
103 dst[newIdx] = src[i];
104 }
105 }

107 template<typename VectorType>
108 static void deCompressVector(const vectorView<const Matrix<VectorType>>& src, vectorView<Matrix<VectorType>,Eigen::RowMajor> dst, std::shared_ptr<compressor> thisPtr)
109 {
110 std::shared_ptr<compressor> srcCompressor;
111 assert(src.getIsCompressed(srcCompressor) == true); // decompressing a decompressed vector is an error. This is to catch bugs
112 assert(srcCompressor.get() == thisPtr.get());
113
114 dst.resize(thisPtr->compressPerm.size(),false,thisPtr);
115 for (size_t i = 0; i < thisPtr->decompressPerm.size(); i++)
116 {
117 int64_t newIdx = thisPtr->decompressPerm[i];
118 dst[newIdx] = src[i];
119 }
120 }
121 compressor(){}

125 size_t getUnCompressedSize(){return compressPerm.size();}
