19constexpr char bitwiseDot(const uint32_t a, const uint32_t b, int dim)
20{
21 // uint32_t prod = a&b;
22 // char ret = 0; //at most 32
23 // for (int i = 0; i < dim; i++)
24 // {
25 // ret += (prod & 1);
26 // prod>>=1;
27 // }
28 // return ret;
29 //Magic code that compiles to popcnt https://stackoverflow.com/questions/109023/count-the-number-of-set-bits-in-a-32-bit-integer
30 uint32_t i = a & b & (dim < 32 ? ((1<<dim) -1) : -1);
31 i = i - ((i >> 1) & 0x55555555); // add pairs of bits
32 i = (i & 0x33333333) + ((i >> 2) & 0x33333333); // quads
33 i = (i + (i >> 4)) & 0x0F0F0F0F; // groups of 8
34 i *= 0x01010101; // horizontal sum of bytes
35 return i >> 24; // return just that top byte (after truncating to 32-bit even when int is wider than uint32_t)
36}

52 numberOperatorCompressor(uint32_t numberOfParticles, uint32_t stateVectorSize)
53 {
54 assert(numberOfParticles <= 32);
55
56
57 compressPerm.resize(stateVectorSize);
58 uint32_t activeCount = 0;
59 uint32_t allOnes = -1;
60 for (uint32_t i = 0; i < stateVectorSize; i++)
61 {
62 bool indexActive = bitwiseDot(i,allOnes,32) == (char)numberOfParticles;
63 if (indexActive)
64 {
65 compressPerm[i] = activeCount;
66 decompressPerm.push_back(i);
67 activeCount++;
68 }
69 else
70 {
71 compressPerm[i] = -1;
72 }
73 }
74 }
